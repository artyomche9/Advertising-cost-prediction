3. Функция для расчета метрик

computeMetrics <- function(predictions, labels, n_sample = NA) {
    
    # Метрика 1
    if (!is.na(n_sample)) {
        delta <- 1.64 * sqrt(labels * (100 - labels) / n_sample)
        m1 <- mean(((labels - predictions) / delta) ^ 2)
    } else m1 <- NA
    
    # Модуль разности предсказанных и фактических значений, 
    # деленный на фактическое значение
    tmp <- abs((labels - predictions) / labels)

    # Метрика 2.1 - среднее по наблюдениям, 
    # для которых прогноз больше (>=) фактического значения
    m2_1 <- mean(tmp[labels - predictions <= 0])
    # Метрика 2.2 - доля наблюдений,
    # для которых прогноз больше фактического значения на 15% и больше
    m2_2 <- mean((labels - predictions) / labels <= -0.15)
    
    # Метрика 2.3 - среднее по наблюдениям, 
    # для которых прогноз меньше фактического значения
    m2_3 <- mean(tmp[labels - predictions > 0])
    # Метрика 2.4 - доля наблюдений,
    # для которых прогноз меньше фактического значения
    m2_4 <- mean(labels - predictions > 0)
    
    tmp <- (labels - predictions) ^ 2
    # Метрика 3.1 - среднеквадратичное отклонение по наблюдениям,
    # для которых прогноз больше (>=) фактического значения
    m3_1 <- sum(tmp[labels - predictions <= 0]) / sum(labels - predictions <= 0)
    # Метрика 3.2 - среднеквадратичное отклонение по наблюдениям,
    # для которых прогноз меньше фактического значения
    m3_2 <- sum(tmp[labels - predictions > 0]) / sum(labels - predictions > 0)

    # Пункт 4 - это не метрики качества модели, поэтому их не считаем
    
    result <- data.frame("Метрика" = c("Метрика 1", "Метрика 2.1",
                                       "Метрика 2.2", "Метрика 2.3",
                                       "Метрика 2.4", "Метрика 3.1",
                                       "Метрика 3.2"),
                         "Значение" = c(m1, m2_1, m2_2, m2_3, 
                                        m2_3, m3_1, m3_2))
    return(result)
}



metrics <- function(variable, model, test = test,
                    test_x = test_x, n_sample = NA) { 
    # x - строка с именем переменной
    # model - модель
    # n_sample - строка с именем переменной, задающей размер выборки
    
    # Отбираем ненулевые рейтинги, чтобы не было деления на 0
    tvr_label <- test[test[, variable] > 0, variable]
    tvr <- xgb.DMatrix(test_x[test[, variable] > 0, ])
    preds <- predict(model, tvr)
    
    n_sample <- ifelse(is.na(n_sample), 
                       NA,
                       test[test[, variable] > 0, n_sample])
    computeMetrics(preds, tvr_label, n_sample = n_sample)
}
